题解 [CodeForces - 527C]Glass Carving

vjudge链接

方法：STL_multiset和set

首先我们可以发现，横着切与竖着切互相是没有影响的。

其实最大面积就是横向的最大间距乘上竖向的最大间距。

如果我们新加一道切痕，而这道切痕只会影响到一个间距，我们只需要找到这个间距，删掉在添加以这条分割线为中心线的左右两个间距即可。

用 setsetset 维护每条切痕的位置。用 multisetmultisetmultiset 维护每个区间的长度，共 444 个。

其优势是可以自动排序，setsetset 可以去重。

查找的方式很简单，找到第一个大于等于此切割点的地方左移一位的位置到此位置便是区间。

具体见代码。

AC记录

Code:

#include<bits/stdc++.h>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}
//x,y表示区间长度，x2,y2表示切割点的位置
multiset<int> x,y;
set<int> x2,y2;
int main () {
	std::ios::sync_with_stdio(false);
	int w,h,n;
	scanf("%d%d%d",&w,&h,&n);
    //初始化
	x.insert(w);
	y.insert(h);  
	x2.insert(0);
	x2.insert(w);
	y2.insert(0);
	y2.insert(h);   
	char a;
	int b;
	for (int i = 1;i <= n;i++) {
		getchar();
        scanf("%c %d",&a, &b);
		if (a == 'V') {
            //找到第一个大于等于b的位置
			multiset<int>::iterator it1,it2,it3;
			it1 = x2.lower_bound(b);
			it2 = it1;
            //左移
			if (it1 != x2.begin()) {
				it1--;
			}   
            //加入该切割点
			x2.insert(b); 
            //区间长度
			int len = *it2 - *it1;
			it3 = x.lower_bound(len);
            //删除
			x.erase(it3);
            //加入区间长度
			x.insert(*it2 - b);
			x.insert(b - *it1); 
		} else {
            //同理
			multiset<int>::iterator it1,it2,it3;
			it1 = y2.lower_bound(b);
			it2 = it1;
			if (it1 != y2.begin()) {
				it1--;
			}   
			y2.insert(b); 
			int len = *it2 - *it1;
			it3 = y.lower_bound(len);
			y.erase(it3);
			y.insert(*it2 - b);
			y.insert(b - *it1);  
		}
		printf("%lld\n",1ll * *x.rbegin() * *y.rbegin());
	} 
	return 0;
}
