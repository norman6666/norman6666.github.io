方法：贪心+模拟

这题我们可以用贪心+模拟解决。

我们将这两个人的马排个序，从大到小排序。

我们利用四个指针，分别是 opu1,opu2,opd1,opd2opu1,opu2,opd1,opd2opu1,opu2,opd1,opd2。

表示如下：

田忌：aaa 数组，另一人：bbb 数组。

我们先关心两边速度最小的马，共三种情况：

①：当 a[opd1]>b[opd2]a[opd1] > b[opd2]a[opd1]>b[opd2]，即田忌的速度最小的马大于另一人的速度最小的马，就可以直接对决，田忌胜，此时 opd1−−,opd2−−opd1--,opd2--opd1−−,opd2−−，ansansans 加上 200200200。

②：当 a[opd1]<b[opd2]a[opd1] < b[opd2]a[opd1]<b[opd2]，即田忌的速度最小的马小于另一人的速度最小的马，这时田忌的速度最小的马的速度小于对方全部的马的速度，所以说这局必输，按照贪心的思路肯定是用这匹马去和对面最快的马对决，此时 opd1−−,opu2++opd1--,opu2++opd1−−,opu2++，ansansans 减上 200200200。

③：当 a[opd1]=b[opd2]a[opd1] = b[opd2]a[opd1]=b[opd2]，即田忌的速度最小的马等于另一人的速度最小的马，分情况考虑速度最大的马：

  • 一：当 a[opu1]<b[opu2]a[opu1] < b[opu2]a[opu1]<b[opu2]，即田忌的速度最快的马小于另一人的速度最快的马，最优方案为田忌用最慢的马与对方最好的马对决，自己用最好的马与对方最慢的马对决，此时收益为 000。而不是两匹最慢的马打平，两匹最快的马打输，这时收益为 −200-200−200。所以此时opd1−−,opu2++opd1--,opu2++opd1−−,opu2++，ansansans 减上 200200200。
  • 二：当 a[opu1]>b[opu2]a[opu1] > b[opu2]a[opu1]>b[opu2]，即田忌的速度最快的马大于另一人的速度最快的马，我们先放着两边两匹最慢的马，处理两边两匹最快的马，田忌胜，此时 opu1++,opu2++opu1++,opu2++opu1++,opu2++，ansansans 加上 200200200。
  • 三：当 a[opu1]=b[opu2]a[opu1] = b[opu2]a[opu1]=b[opu2]，即田忌的速度最快的马等于另一人的速度最快的马，这种情况比较特殊。当 a[opd1]<b[opu2]a[opd1] < b[opu2]a[opd1]<b[opu2]，即田忌的速度最慢的马小于另一人的速度最快的马，此时我们可以将田忌的速度最慢的马小于另一人的速度最快的马对决，田忌现在用慢的去输一局，快的马也会赢回来，这波不亏也不赚，但是导致了错位，使后面更有可能会赢（典型例子：田忌3 2 1，另一人3 2 1，此时收益为 200200200）。不满足这条件时，我们才让他们打平局，即 opd1−−,opd2−−,opu1++,opu2++opd1--,opd2--,opu1++,opu2++opd1−−,opd2−−,opu1++,opu2++。

#include<iostream>
#include<algorithm>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}
const int N = 1010;
int a[N],b[N];

bool cmp (int a,int b) {
	return a > b;
}
int main () {
	std::ios::sync_with_stdio(false);
	while(true) {
		int n;
		n = read();
		if (n == 0) {
			break;
		}
		for (int i = 1;i <= n;i++) {
			a[i] = read();
		}
		for (int i = 1;i <= n;i++) {
			b[i] = read();
		}
		sort(a+1,a+1+n,cmp);
		sort(b+1,b+1+n,cmp);
		int ans = 0;
		int opd1 = n,opd2 = n,opu1 = 1,opu2 = 1;
		for(int i = 1;i <= n;i++) {
			if (a[opd1] > b[opd2]) {
				opd1--,opd2--;
				ans += 200;
			} else if (a[opd1] < b[opd2]) {
				opd1--,opu2++;
				ans -= 200;
			} else {
				if (a[opu1] < b[opu2]) {
					opd1--,opu2++;
					ans -= 200;
				} else if (a[opu1] > b[opu2]){
					opu1++,opu2++;
					ans += 200;
				} else {
					if (a[opd1] < b[opu2]) {
						opd1--,opu2++;
						ans -= 200;
					} else {
						opd1--,opd2--,opu1++,opu2++;
					}
				}
			}
		}
		cout << ans << endl;
	}
	return 0;
}
