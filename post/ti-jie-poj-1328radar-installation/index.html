题解 [POJ 1328]Radar Installation

方法：贪心

这题我们用贪心解决。

我们可以将题意转换一下，对于一个二维坐标，记录能满足在 xxx 轴放雷达的最小位置和最大位置，原题意就转换成了给你一些线段，求用最少的点覆盖所有的线段的点数，即每个线段上都至少要有一个点。

最小位置和最大位置的求法：

利用勾股定理：l=x−d2−y2,r=x+d2−y2(d≥y)l = x - \sqrt{d^{2} - y^{2}},r = x + \sqrt{d^{2} - y^{2}}(d ≥ y)l=x−d2−y2​,r=x+d2−y2​(d≥y)。

当 d<yd < yd<y，这个坐标一定不会扫到，我们这时就可以记录无解了。

用 nownownow 表示此时处理到的位置，用 ansansans 记录答案。

按右端点排序后，将 nownownow 初始值为第一个的右端点，这样一定能覆盖到第一个坐标。

初始化时 now=3now = 3now=3，那么我们的 1、2、3、41、2、3、41、2、3、4 都可以被覆盖到，我们接下来枚举到第一个左端点大于 nownownow 的位置停止（样例中的位置为 555），更新 nownownow，ansansans 加一，如此这样循环。

#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}
const int N = 1010;

struct node {
	double x,y;
}a[N],b[N];

bool cmp (node a,node b) {
	return a.y < b.y;
}

int main () {
	std::ios::sync_with_stdio(false);
	int n,d;
	int t = 0;
	while (true) {
		cin >> n >> d;
		if (n == 0 && d == 0) {
			break;
		}
		t++;
		int cnt = 0;
		bool _ = true;
		for (int i = 1;i <= n;i++) {
			cin >> a[i].x >> a[i].y;
			//覆盖不到，y > d
			if (abs(a[i].y) > d) {
				_ = false;
				continue;
			}
			//记录左端点和右端点 
			cnt++;
			b[cnt].x = a[i].x - sqrt(pow(d,2.0) - pow(a[i].y,2.0));
			b[cnt].y = a[i].x + sqrt(pow(d,2.0) - pow(a[i].y,2.0));
		}
		//无法覆盖 
		if (_ == false) {
			cout << "Case " << t << ": -1" << endl;
			continue;
		} 
		//按右端点排序 
		sort(b+1,b+1+cnt,cmp);
		//初始值now 
		double now = b[1].y;
		int ans = 1;
		for (int i = 2;i <= cnt;i++) {
			//允许的一个误差1e-8，左端点大于now时 
			if (b[i].x - now > 1e-8) {
				//更新 
				now = b[i].y;
				ans++;
			}
		}
		//输出 
		cout << "Case " << t << ": " << ans << endl;
	}
	return 0;
}
