方法：贪心

这题我们可以用贪心解决。

对于选的决策，当队首不等于队尾的时候很好办，选最小的行。

我们的主要问题出在当队首等于队尾的时候选哪边？

好比这样一个序列：CBAC，字典序最小的排法为CABC。

如果我们碰见相同的时候，假如我们第一次选了队首的C，则我们的答案将变成CBAC。

所以相同时不可以随意选一个。

当我们遇见队首和队尾相同时，继续往中间比较，直至出现两个不同的字符，再取小的那边，保证字典序最小。

具体见代码。

#include<iostream>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}

const int N = 2010;
char a[N];
char ans[N];
int main () {
	std::ios::sync_with_stdio(false);
	int n;
	cin >> n;
	char x;
	for (int i = 1;i <= n;i++) {
		cin >> x;
		//存入数组 
		a[i] = x;
	} 
	int cnt = 0;
	int l = 1,r = n;
	while(l <= r) {
		//相等的棘手情况 
		if (a[l] == a[r]) {
			int i = l,j = r;
			//过滤两端的字符相等的情况  
			while(a[i] == a[j] && i < j) {
				i++,j--;
			}
			//这里忽略的一个细节，当 i > j 时，表示这是一个回文字符串
			//代码这里就直接进入else了，选队尾
			//其他的更新都不难了 
			if (a[i] < a[j]) {
				cnt++;					
				ans[cnt] = a[l];
				l++;
			} else {
				cnt++;
				ans[cnt] = a[r];
				r--;
			}
		} else {
			//其他的更新都不难了 
			if (a[l] < a[r]) {
				cnt++;
				ans[cnt] = a[l];
				l++;
			} else {
				cnt++;
				ans[cnt] = a[r];
				r--;
			}
		}
	}
	//奇怪的输出要求 
	for (int i = 1;i <= n;i++) {
		if (i % 80 == 0) {
			cout << ans[i] << endl;
		} else {
			cout << ans[i];
		}
	}
	cout << endl;
	return 0;
}
