题解 [Light OJ 1043]Triangle Partitioning

方法：二分

这题我们可以用二分解决。

我们设 ad=xad = xad=x。

我们发现 xxx 越大，SADE:SBDECS_{ADE}:S_{BDEC}SADE​:SBDEC​ 的值越大，满足单调性。

我们通过二分我们枚举 adadad 的长度，求得的比值与给定的比值对比二分即可。

∵AD:AB=AE:AC∵AD:AB = AE:AC∵AD:AB=AE:AC，

∴AE=AC∗AD/AB∴AE = AC * AD / AB∴AE=AC∗AD/AB。

海伦公式：三条边长 a,b,ca,b,ca,b,c，p=(a+b+c)/2p = (a+b+c)/2p=(a+b+c)/2，S△=p∗(p−a)∗(p−b)∗(p−c)S_{△} = \sqrt{p*(p-a)*(p-b)*(p-c)}S△​=p∗(p−a)∗(p−b)∗(p−c)​。

具体见代码。

#include<bits/stdc++.h>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}
inline void write(ll x){
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}

const double eps = 1e-8;
double ab,ac,bc,k;

double run (double a,double b,double c) {
	double p = (a + b + c) / 2;
	double s = sqrt(p*(p-a)*(p-b)*(p-c));
	return s;
}

double get(double mid) {
	//求ad,ae,de 
	double ad = mid;
	double ae = ac * ad / ab;
	double de = ad * bc / ab;
	//run(ad,ae,de)->S_ade
	//(run(ab,ac,bc) - run(ad,de,ae)->S_abc-S_ade=S_bcde
	return run(ad,ae,de) / (run(ab,ac,bc) - run(ad,de,ae));
}
int main () {
	int t = read(),cnt = 0;
	while(t--) {
		scanf("%lf%lf%lf%lf",&ab,&ac,&bc,&k);
		double l = 0,r = ab,best = -1;
		//二分 
		while(fabs(r - l) > eps) {
			double mid = (l + r) / 2;
			//大于比值 
			if (get(mid) >= k) {
				best = mid;
				r = mid;
			} else {
				l = mid;
			}
		}
		cnt++;
        //保留8位
		printf("Case %d: %.8f\n",cnt,best);
	} 
	return 0;
}
