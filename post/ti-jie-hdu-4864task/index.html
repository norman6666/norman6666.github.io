方法：贪心

这题我们可以用贪心解决。

这里的一些任务，时间这些数据转换到一个二维坐标系中，当机器位于任务的右上方（即 任务x≤机器x任务_{x} ≤ 机器_{x}任务x​≤机器x​ 且 任务y≤机器y任务_{y} ≤ 机器_{y}任务y​≤机器y​）我们就可以将这个机器和任务匹配。

由于一个任务的价值是 500∗x+2∗y500 * x + 2 * y500∗x+2∗y，xxx 的占比较重，所以我们优先按 xxx 从大到小排序，当 xxx 相等时按 yyy 从大到小排序，得到的利益最大。

排序后我们的两个数组都是满足单调性的。

我们可以维护两个指针，用 iii 来枚举每个任务，用 jjj 来枚举现在机器的横坐标大于等于此任务的横坐标的所有机器，用一个桶来装，以便于我们之后的查找，这样我们的 jjj 不用再次查找，只需要向后枚举即可，因为我们的数组排序后都是满足单调性的。

每次处理完桶后，就从此任务的纵坐标枚举到最大值 100100100，找到一个最小的 yyy，与其机器匹配。找到最小的 yyy 的原因是 yyy 更大的可以去匹配一个 yyy 更大的任务。

具体见代码。

#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}

const int N = 100010;
struct node {
	ll x,y;
}a[N],b[N]; 

const int M = 110;
ll cnt[M]; 

bool cmp (node a,node b) {
	if (a.x == b.x) {
		return a.y > b.y;
	} else {
		return a.x > b.x;
	}
}
int main () {
	std::ios::sync_with_stdio(false);
	int n,m;
	while(~scanf("%d%d",&n,&m)) {
		for (int i = 1;i <= n;i++) {
			a[i].x = read(),a[i].y = read();
		}
		for (int i = 1;i <= m;i++) {
			b[i].x = read(),b[i].y = read();
		}
		//排序 
		sort(a+1,a+1+n,cmp);
		sort(b+1,b+1+m,cmp);
		int j = 1;
		memset(cnt,0,sizeof(cnt));
		ll ans = 0,counter = 0;
		//指针i,j,答案ans和计数器counter，桶cnt 
		for (int i = 1;i <= m;i++) {
			//用 j 来枚举现在机器的横坐标大于等于此任务的横坐标的所有机器 
			//用桶存 
			while (j <= n && a[j].x >= b[i].x) {
				cnt[a[j].y]++;
				j++;
			} 
			//找最小的y 
			for (int l = b[i].y;l <= 100;l++) {
				if (cnt[l] != 0) {
					counter++;
					ans += b[i].x * 500 + b[i].y * 2;
					//记得要减掉 
					cnt[l]--;
					break;
				}
			}
		}
		cout << counter << ' ' << ans << endl;
	}
	return 0;
}
