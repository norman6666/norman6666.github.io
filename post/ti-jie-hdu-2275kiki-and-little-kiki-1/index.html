multiset<int> x;

int main () {
	int n;
	while(~scanf("%d",&n)) {
		int b;
		for (int i = 1;i <= n;i++) {
			scanf("%s %d",&a,&b);
			if (a[1] == 'u') {
				//插入 
				x.insert(b);
			} else {
				//找到第一个大于 M 的位置 
				multiset<int>::iterator it = x.upper_bound(b);
				//如果不是第一个 
				if (it != x.begin()) {
					//左挪一位 
					it--;
					printf("%d\n",*it);
					//删掉 
					x.erase(it);
				} else {
					//没找到 
					puts("No Element!");
				}
			}
		}

		x.clear();
	}
