方法一：手写压位（暴力优化）

众所周知，intintint 类型是 323232 位整数。

由于我们开数组的话每个 intintint 只有两个意义：在与不在，造成了很多浪费，所以我们可以通过二进制来表示。

所以一个 intintint 是一个 323232 位二进制数，可以表示 323232 种状态。

那么从暴力的 10000∗100010000*100010000∗1000 我们可以压缩为 10000∗4010000*4010000∗40。（40∗32>100040*32>100040∗32>1000）

其 addaddadd 核心函数代码：

void add(int x,int y) {
	//将 y/32 集合内的第 y % 32 的地方标记为 1 
	a[x][y / 32] |= (1 << (y % 32));
}

#include<bits/stdc++.h>
using namespace std;
//考虑unsigned long long
typedef long long ll;
inline int read(){
	int z=0,x=1;char c=getchar();
	while(c>'9'||c<'0'){if(c=='-') x=-x; c=getchar();}
	while(c>='0'&&c<='9') z=z*10+c-'0',c=getchar();
	return z*x;
}
inline void write(ll x){
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
//x 属于 y 集合 
int a[10010][40];
void add(int x,int y) {
	//将 y/32 集合内的第 y % 32 的地方标记为 1 
	a[x][y / 32] |= (1 << (y % 32));
}
 
int main () {
	std::ios::sync_with_stdio(false);
	int n = read();
	for (int i = 1;i <= n;i++) {
		int t = read();
		int b;
		for (int j = 1;j <= t;j++) {
			b = read();
			//添加 
			add(b,i);
		}
	}
	int q = read();
	int x,y;
	while(q--) {
		x = read(),y = read();
		bool _ = false;
		//询问 x y 是否属于同一集合 
		for (int i = 0;i < 40;i++) {
			if (a[x][i] & a[y][i]) {
				_ = true;
				puts("Yes");
				break;
			}	
		}
		if (_ == false) {
			puts("No");
		}
	}
	return 0;
}
